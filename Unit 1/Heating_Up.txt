public class MakingPopcorn
{
    public static void main(String[] args)
    {
        System.out.println("3 Ways to Heat Anything");
        System.out.println("Each of the 3 methods below is an example of the three ways that heat can be transferred.\n");
        
        System.out.println("Conduction");
        System.out.println("Conduction is heat transfer through matter. Metals conduct heat well, but air does not. This is a direct contact type of heat transfer.\n");
        
        System.out.println("Convection");
        System.out.println("Convection is heat transfer by the movement of mass from one place to another. It can take place only in liquids and gases. Heat gained by conduction or radiation is moved by convection.\n");
        
        System.out.println("Radiation");
        System.out.println("Radiation can transfer heat through the relative emptiness of space. All other forms of heat transfer require motion of molecules like air or water to move heat.\n");
    }
}
